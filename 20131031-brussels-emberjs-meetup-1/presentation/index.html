Controllers provide an interface of the model to the view.


App.PersonController =
  Ember.ObjectController.extend({
    fullName: function() {
      return (
        this.get('firstName')
        + " "
        + this.get('lastName'));
    }.property('firstName', 'fullName')
});

The Router transitions between a collection of routes, each representing an application state and identified by a URL.

Each Route tells the template which model it should display.

The router interface is similar to the RoR router.


App.Router.map(function() {
  this.route("about", { path: "/about" });
  this.route("favorites", { path: "/favs" });
  this.resource('users', function() {
    this.route('new');
  });
});
